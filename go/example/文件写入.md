// os.WriteFile 取代ioutil.WriteFile 每次会覆盖文件内容，如果文件不存在会创建
func writeByOSWriteFile(filename string) {
	content := []byte("测试1\n测试2\n")
	err := os.WriteFile(filename, content, 0644)
	if err != nil {
		panic(err)
	}
}

- os.O_WRONLY   只写
- os.O_CREATE   创建文件
- os.O_RDONLY   只读
- os.O_RDWR     读写
- os.O_TRUNC    清空
- os.O_APPEND   追加
*/

// io.WriteString 可以追加到文件
func writeByIOWriteString(filename string) {
	var writeString = "writeByIOWriteString\n"
	var file *os.File
	var err error
	if checkFileIsExist(filename) { //如果文件存在
		file, err = os.OpenFile(filename, os.O_APPEND|os.O_WRONLY, 0644) //打开文件
		fmt.Println("文件存在,追加内容")
	} else {
		file, err = os.Create(filename) //创建文件
		fmt.Println("文件不存在")
	}
	if err != nil {
		panic(err)
	}
	defer file.Close()

	n, err := io.WriteString(file, writeString) //写入文件(字符串)

	}

}

// file.Write 可以追加到文件

// writer.WriteString 可以追加到文件
func writeByBufioWriteString(filename string) {
	var writeString = "writeByBufioWriteString\n"
	var file *os.File
	var err error
	if checkFileIsExist(filename) { //如果文件存在
		file, err = os.OpenFile(filename, os.O_APPEND|os.O_WRONLY, 0644) //打开文件
		fmt.Println("文件存在,追加内容")
	} else {
		file, err = os.Create(filename) //创建文件
		fmt.Println("文件不存在")
	}
	if err != nil {
		panic(err)
	}
	defer file.Close()

	writer := bufio.NewWriter(file) //创建新的 Writer 对象
	n, _ := writer.WriteString(writeString)
	fmt.Printf("写入 %d 个字节", n)
	writer.Flush()
}

func writeByBufioNewWriter(filename string) {

	if checkFileIsExist(filename) { //如果文件存在
		file, err = os.OpenFile(filename, os.O_APPEND|os.O_WRONLY, 0666) //打开文件
		fmt.Println("文件存在")
	} else {
		file, err = os.Create(filename) //创建文件
		fmt.Println("文件不存在")
	}
	if err != nil {
		panic(err)
	}
	defer file.Close()
	writer := bufio.NewWriter(file) //创建新的 Writer 对象
	n4, err := writer.WriteString(msgString)
	if err != nil {
		panic(err)
	}
	fmt.Printf("写入 %d 个字节n", n4)
	writer.Flush()
}

func checkFileIsExist(filename string) bool {
	if _, err := os.Stat(filename); os.IsNotExist(err) {
		return false
	}
	return true
}
